Mixing Chinese and English Fonts

Add into file /etc/fonts/local.conf

<?xml version="1.0"?>
<!DOCTYPE fontconfig SYSTEM "fonts.dtd">
<fontconfig>
  <alias>
  <family>serif</family>
    <prefer><family>Linux Libertine Display O</family></prefer>
  </alias>
  <alias>
  <family>sans-serif</family>
    <prefer><family>Cantarell</family></prefer>
    <prefer><family>WenQuanYi Micro Hei</family></prefer>
  </alias>
  <alias>
  <family>sans</family>
    <prefer><family>Cantarell</family></prefer>
    <prefer><family>WenQuanYi Micro Hei</family></prefer>
  </alias>
  <alias>
  <family>monospace</family>
    <prefer><family>Monaco</family></prefer>
  </alias>
  
  <match target="font">
  <test name="weight" compare="more">
  <const>medium</const>
  </test>
  <edit name="autohint" mode="assign">
  <bool>false</bool>
  </edit>
  </match>
</fontconfig>
